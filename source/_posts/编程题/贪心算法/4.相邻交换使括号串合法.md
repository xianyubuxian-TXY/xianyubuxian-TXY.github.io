# 2.题目解析
- **贪心**：
	- sum=0
	- 遇到'('，sum+1 ; 遇到'(',sum-1
	- 当sum<0时，说明存在无效')',则之后每遇到'(',就用最远的无效')'进行匹配，正好等于 abs(sum)，**交换次数+=abs(sum)**

# 3.代码实现
```cpp
#include<iostream>
using namespace std;

long long MinSwap(string& s,int n){
    long long res=0;
    long long sum=0;
    for(int i=0;i<n;i++){
        if(s[i]=='('){
            if(sum<0) res+=abs(sum);
            sum++;
        }else{
            sum--;
        }
    }
    if(sum!=0) return -1;
    return res;
}

int main(){
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
        int n;
        cin>>n;
        string s;
        cin>>s;
        printf("%lld\n",MinSwap(s,n));
    }
    
    return 0;
}
```
