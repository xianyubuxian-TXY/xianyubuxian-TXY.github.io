#include <system_error>
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * pick candy
     * @param arr int整型vector the array
     * @return int整型
     */
    int candy(vector<int>& arr) {
        int len=arr.size();
        // 贪心：初始时，每个孩子都先分配1个糖果，然后在进行两次扫描 ——> 从左到右扫描、从又到左扫描
        // 因为一个孩子最多有两个相邻孩子，所以进行两次扫描
        // 1.从左到右扫描：如果左边得分<右边得分，则 右边糖果数=左边糖果数+1 ——> 保证一个孩子右边孩子比他得分高，则糖果多
        // 2.从右到左扫描：如果右边得分<左边得分，则 左边糖果数=max(左边糖果,右边糖果数+1) ——> 保证一个孩子左边孩子比他得分高，则糖果多，注意用max，避免“破坏左侧约束”
        vector<int> suger_nums(len,1);
        
        // 从左向右扫描
        int index=0;
        for(int i=1;i<len;i++){
            if(arr[index]<arr[i]) suger_nums[i]=suger_nums[index]+1;
            index=i;
        }

        // 从右到左扫描
        index=len-1;
        for(int i=len-2;i>=0;i--){
            if(arr[index]<arr[i]) suger_nums[i]=max(suger_nums[i],suger_nums[index]+1);
            index=i;
        }

        // 计算总和
        int sum=0;
        for(auto it: suger_nums){
            sum+=it;
        }
        return sum;
    }
};
