```cpp
class Solution {
public:
    int findMaximizedCapital(int k, int w, vector<int>& profits, vector<int>& capital) {
        int len=profits.size();
        // {所需本金, 利润}
        vector<pair<int,int>> arr;
        for(int i=0;i<len;i++){
            arr.push_back({capital[i],profits[i]});
        }
        // 按所需本金从小到大排序
        sort(arr.begin(),arr.end());    

        priority_queue<int> pq; // 大根堆，存当前能做项目的利润
    
        int index=0;
        // 最多做 k 个项目
        while(k>0){
            // 把当前所有能做的项目都加入堆中
            while(index<len && arr[index].first<=w){
                pq.push(arr[index].second);
                index++;
            }

            // 没有能做的项目了
            if (pq.empty()) break;

            // 选择利润最大的项目
            w+=pq.top();
            pq.pop();

            k--;
        }
        return w;
    }
};
```
