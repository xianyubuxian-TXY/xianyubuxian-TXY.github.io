# 2.题目解析
- 用uSet（unordered_set）存储nums中的所有值，对于每个值，依次向“左右扩展”，如果扩展后的值也在uSet中，则连续长度+1

# 3.代码实现
```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        // 存储val
        unordered_set<int> uSet;
        int len=nums.size();

        for(int i=0;i<nums.size();i++){
            uSet.insert(nums[i]);
        }

        // 对于nums中的每个数，将其“向左右扩展”，看是否存在与uSet中
        // 如果存在，则连续长度+1，同时将其从uSet中去掉，表示被访问过了
        int maxlen=0;
        for(int i=0;i<nums.size();i++){
            int tmplen=1;
            int cur_val=nums[i];
            if(!uSet.count(cur_val)) continue;  // 不存在，说明访问过了
            uSet.erase(cur_val);    // 从uSet中移除，避免重复访问
            int left_val=cur_val-1,right_val=cur_val+1;
            // 向左寻找
            while(uSet.count(left_val)){
                tmplen++;
                uSet.erase(left_val);
                left_val--;
            }

            // 向右寻找
            while(uSet.count(right_val)){
                tmplen++;
                uSet.erase(right_val);
                right_val++;
            }

            // 更新长度
            maxlen=max(maxlen,tmplen);
        }
        return maxlen;
    }
};
```
