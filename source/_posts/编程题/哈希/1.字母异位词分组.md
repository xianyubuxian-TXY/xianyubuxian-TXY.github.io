# 3.代码实现
```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<string,vector<string>> uMap;
        for(int i=0;i<strs.size();i++){
            string s=strs[i];
            sort(s.begin(),s.end());
            uMap[s].push_back(strs[i]);
        }
        vector<vector<string>> res;
        for(auto& it:uMap){
            res.push_back(it.second);
        }
        return res;
    }
};
```
