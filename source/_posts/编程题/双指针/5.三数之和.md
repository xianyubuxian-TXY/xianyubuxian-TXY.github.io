# 3.代码实现
```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        int len=nums.size();
        sort(nums.begin(),nums.end());
        
        vector<vector<int>> res;
        if(nums[0]>0 || nums[len-1]<0) return res;
        // 枚举每一个数，再用双指针去寻找剩下两个数
        // 重点！！：去重
        for(int i=0;i<len;i++){
            // 剪枝
            if(nums[i]>0) break;
            
            int left=i+1,right=len-1;
            
            // 第1个数去重
            if(i>0 && nums[i]==nums[i-1]) continue;

            while(left<right){
                
                int sum=nums[left]+nums[i]+nums[right];
                if(sum==0){
                    // 找到一组
                    res.push_back({nums[left],nums[i],nums[right]});
                    
                    // 第2个数去重
                    while(left+1<right && nums[left+1]==nums[left]) left++;

                    // 第3个数去重
                    while(right-1>left && nums[right-1]==nums[right]) right--;

                    // 双指针同时向内收缩
                    left++;
                    right--;
                }else if(sum<0){
                    // sum<0：left向内收缩
                    left++;
                }else{
                    // sum>0：right向内收缩
                    right--;
                }
            }
        }
        return res;
    }
};
``` 
