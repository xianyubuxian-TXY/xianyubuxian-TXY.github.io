```cpp
#include <climits>
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param S string字符串 
     * @param T string字符串 
     * @return string字符串
     */
    string minWindow(string S, string T) {
        // write code here

        int len1=S.size();
        int len2=T.size();
        if(len1<len2 || len2==0) return "";
        unordered_map<char,int> needs;  // 字母 ——> 需要个数
        int missing=T.size();           // 总共需要的字母数量
        for(int i=0;i<len2;i++){
            needs[T[i]]++;
        }

        pair<int,int> res={0,10001};
        int left=0,right=0; // 双指针,[left,right-1]为满足的区间
        while(right<len1){
            // right向右扩展，直到包含T的所有字符
            while(missing>0 && right<len1){
                char c=S[right];
                // S[right]是T中的字符
                if(needs.find(c)!=needs.end()){
                    if(needs[c]>0) missing--;   // 仍属于需要的字母
                    needs[c]--;
                }
                right++;
            }

            // left向右放缩，直到不满足包含T的所有字符
            while(missing==0 && left<len1){
                char c=S[left];
                if(needs.find(c)!=needs.end()){
                    needs[c]++;
                    if(needs[c]>0){  // 有字符需要，不在包含T中所有字符
                        missing++;
                        // S[left,right-1]是有效区间，看是否需要更新区间
                        if(right-left < res.second-res.first+1){
                            res.first=left;
                            res.second=right-1;
                        }

                    }
                }
                left++;
            }
        }

        if(res.second==10001) return "";
        return S.substr(res.first,res.second-res.first+1);
        
    }
};
```
