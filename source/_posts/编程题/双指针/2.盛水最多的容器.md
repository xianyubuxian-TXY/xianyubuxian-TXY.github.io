# 2.解题思路
- 关键是题解题目：选择两个坐标轴x1，x2后，它们之间的“柱子”就不用管了，忽略掉，是height[x1],height[x2]与x轴组成的容器 ——> 计算最大面积

# 3.代码实现
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param height int整型vector 
     * @return int整型
     */
    int maxArea(vector<int>& height) {
        // write code here
        int len=height.size();
        // 注意理解题目：选择两个坐标轴x1，x2后，它们之间的“柱子”就不用管了，忽略掉，是height[x1],height[x2]与x轴组成的容器 ——> 计算最大面积
        if(len<=1) return 0;
        int left=0,right=len-1; // 双指针
        int res=0;

        while(left<right){
            // 计算当前面积
            int min_height=min(height[left],height[right]);
            res=max(res,min_height*(right-left));
            
            // 移动矮的一边，这样才可能让面积增加
            if(height[left]<height[right]) left++;
            else right--;
        }

        return res;
    }
};
```
