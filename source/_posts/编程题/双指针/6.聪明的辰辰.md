# 3.代码实现
```cpp
#include<iostream>
using namespace std;
#include<vector>
#include<algorithm>

int main(){
    int N,T;
    cin>>N>>T;
    vector<int> arr(N);
    for(int i=0;i<N;i++){
        cin>>arr[i];
    }
    sort(arr.begin(),arr.end());

    // pre[i]：表示arr[1..i]中最长满足条件的连续子序列的长度（从左往右）
    vector<int> pre(N+1,0);
    int left=1,right=1;
    for(;right<=N;right++){
        pre[right]=pre[right-1];

        while(arr[right-1]-arr[left-1]>T){
            // 不满足，则left指针向右收缩
            left++;
        }
        pre[right]=max(pre[right],right-left+1);
    }

    // suf[i]：表示arr[i..N]中最长满足条件的连续子序列的长度（从右往左）
    vector<int> suf(N+2,0);
    left=N,right=N;
    for(;left>=1;left--){
        suf[left]=suf[left+1];

        while(arr[right-1]-arr[left-1]>T){
            // 不满足，则right指针向左收缩
            right--;
        }

        suf[left]=max(suf[left],right-left+1);
    }

    // 枚举pre[i]、end[i+1],找到最大两者和
    int res=0;
    for(int i=1;i<=N;i++){
        res=max(res,pre[i]+suf[i+1]);
    }

    cout<<res;
    return 0;
}
```
