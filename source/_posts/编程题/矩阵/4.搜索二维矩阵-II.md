# 3.代码实现
```cpp
class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int rowSize=matrix.size();
        int colSize=matrix[0].size();
        int i=0,j=colSize-1;
        while(i<rowSize && j>=0){
            if(matrix[i][j]==target) return true;
            else if(matrix[i][j]<target){
                // 小于，则行数+1
                i++;                
            }else{
                // 大于，则列数-1
                j--;
            }
        }
        return false;
    }
};
```
