```cpp
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {        
        int rowSize=matrix.size();
        int colSize=matrix[0].size();
        
        // 第一行是否原本就含0，需要整行变0
        bool firstRowZero=false;
        // 第一列是否原本就含0，需要整列变0
        bool firstColZero=false;

        // 检查第一行是否原本就含0
        for(int i=0;i<colSize;i++){
            if(matrix[0][i]==0){
                firstRowZero=true;
                break;
            }
        }

        // 检查第一列是否原本就含0
        for(int i=0;i<rowSize;i++){
            if(matrix[i][0]==0){
                firstColZero=true;
                break;
            }
        }

        // 如果某个位置为0，用第一行、第一列记录要置为0的行、列
        for(int i=1;i<rowSize;i++){
            for(int j=1;j<colSize;j++){
                if(matrix[i][j]==0){
                    matrix[0][j]=0;
                    matrix[i][0]=0;
                }    
            }
        }

        for(auto &row:matrix){
            for(auto &it:row){
                cout<<it<<" ";
            }
            cout<<endl;
        }

        // 根据在第一行、第一列标记的0，进行置0
        // 注意！：这里必须从i=1开始，如果从i=0开始，则当matrix[0][0]=0时，会将第一列全置0，从而影响对第一列的检查
        for(int i=1;i<colSize;i++){
            if(matrix[0][i]==0){
                for(int j=0;j<rowSize;j++){
                    matrix[j][i]=0;
                }
            }
        }


        for(int i=1;i<rowSize;i++){
            if(matrix[i][0]==0){
                for(int j=0;j<colSize;j++){
                    matrix[i][j]=0;
                }
            }
        }
        cout<<endl;
        for(auto &row:matrix){
            for(auto &it:row){
                cout<<it<<" ";
            }
            cout<<endl;
        }

        // 最后看是否需要对第一行、第一列整体置0
        if(firstRowZero){
            for(int i=0;i<colSize;i++) matrix[0][i]=0;
        }
        if(firstColZero){
            for(int i=0;i<rowSize;i++) matrix[i][0]=0;
        }

    }
};
```
