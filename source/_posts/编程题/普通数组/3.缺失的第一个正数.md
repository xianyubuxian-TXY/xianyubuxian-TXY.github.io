# 3.代码实现
```cpp
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int len=nums.size();
        for(int i=0;i<len;i++){

            // 注意：必须使用while循环 ——> 比如 [3,4,-1,1] ,当 swap(4,1)后，4的位置正确了，而1的位置没有正确，所以必须继续交换
            // nums[i]!=nums[nums[i]-1]：避免进入死循环
            while(nums[i]>=1 && nums[i]<=len && nums[i]!=nums[nums[i]-1]){
                swap(nums[i],nums[nums[i]-1]);
            }
        }

        int res=-1;
        for(int i=0;i<len;i++){
            if(nums[i]!=i+1){
               res=i+1;
               break; 
            }
        }
        // 处理nums中所有正数都到齐的情况
        if(res==-1) return len+1;
        return res;
    }
};
```
