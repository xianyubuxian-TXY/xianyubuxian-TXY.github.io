class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int len=nums.size();
        int preMul=1;
        int sufMul=1;
        vector<vector<int>> Muls(len,vector<int>(2,1));
        for(int i=0,j=len-1;i<len-1 && j>0;i++,j--){
            preMul*=nums[i];
            Muls[i+1][0]=preMul;
            sufMul*=nums[j];
            Muls[j-1][1]=sufMul;
        }

        vector<int> res;
        res.push_back(Muls[0][1]);
        for(int i=1;i<len-1;i++){
            res.push_back(Muls[i][0]*Muls[i][1]);
        }
        res.push_back(Muls[len-1][0]);
        return res;
    }
};
