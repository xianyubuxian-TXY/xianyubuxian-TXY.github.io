# 3.代码实现
```cpp
#include <climits>
#include <iostream>
using namespace std;
#include<vector>

int main() {
    int n,k;
    cin>>n>>k;
    
    // 注意：列数要2*n ， 因为 ai可以取 -200000000，所以不能初始化为0
    // 也不可以初始化为LONG_LONG_MIN，以为如果是负数时，会发生溢出
    // 300*（-2e9）=-6e11
    long long ENG=-4e13;
    vector<vector<long long>> arr(n+1,vector<long long>(2*n+1,ENG));
    for(int i=1;i<=n;i++){
        for(int j=1;j<=(2*i-1);j++){
            cin>>arr[i][j];
        }
    }

    // 注意：列数要2*n
    vector<vector<long long>> dp(n+1,vector<long long>(2*n+1,ENG));
    dp[1][1]=arr[1][1];

    for(int i=2;i<=n;i++){
        for(int j=1;j<=(2*i-1);j++){
            // 注意：第i-1行，最多有2*(i-1)-1列
            int limit=2*(i-1)-1;
                        
            // 1.dp[i][j]来自dp[i-1][j]
            if(j<=limit) dp[i][j]=max(dp[i][j],dp[i-1][j]);

            // 2.dp[i][j]来自dp[i-1][j-1]
            if(j-1>=1 && j-1<=limit) dp[i][j]=max(dp[i][j],dp[i-1][j-1]);

            // 3.dp[i][j]来自dp[i-1][j-2]
            if(j-2>=1 && j-2<=limit) dp[i][j]=max(dp[i][j],dp[i-1][j-2]);

            dp[i][j]+=arr[i][j];
        }
    }

    // 第n行中，列数有效范围 n-k<=j<=n+k
    // 还需要注意，n-k可能为0，n+k可能为2n
    int left=max(1,n-k),right=min(2*n-1,n+k);
    long long res=ENG;
    for(int i=left;i<=right;i++){
        res=max(res,dp[n][i]);
    }

    printf("%lld",res);
    return 0;
}
// 64 位输出请用 printf("%lld")
```
