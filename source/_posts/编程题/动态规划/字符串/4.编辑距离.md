```cpp
#include <climits>
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param str1 string字符串 
     * @param str2 string字符串 
     * @return int整型
     */
    int editDistance(string str1, string str2) {
        // write code here
        int len1=str1.size();
        int len2=str2.size();
        // dp[i][j] 表示把 str1 的前 i 个字符变成 str2 的前 j 个字符的最少操作数
        vector<vector<int>> dp(len1+1,vector<int>(len2+1,INT_MAX));
        // 初始化
        for(int i=0;i<=len1;i++) dp[i][0]=i;    // 把str1前i个字符删掉
        for(int j=0;j<=len2;j++) dp[0][j]=j;    // 在str1中插入str2的前j个字符

        for(int i=1;i<=len1;i++){
            for(int j=1;j<=len2;j++){
                // str1的第i个字符与str2的第j个字符相同，继承前面
                if(str1[i-1]==str2[j-1]){
                    dp[i][j]=dp[i-1][j-1];
                }else{
                    /* 因为是将str1变为str2，所以在不相同时，有三种演化而来的情况
                        - 1.在dp[i-1][j-1]的前提下，将str1的第i个字符变为str2的第j个字符（修改字符）
                        - 2.在dp[i][j-1]的前提下，在str1中i+1插入str[j]（插入字符）
                            - dp[i][j-1]：表示str1[1..i]=str2[1..j-1]了，现在要将str2扩展到j，则str1也要插入str2[j]
                        - 3.在dp[i-1][j]的前提下，将str1中第i个字符删除（删除字符）
                            - dp[i-1][j]：表示str1[1..i-1]=str2[1..j],现在要将str1扩展到i，而str2已经是j不可以动了，则需要删除str1[i]
                    */
                    dp[i][j]=min(dp[i-1][j-1]+1,dp[i][j-1]+1);
                    dp[i][j]=min(dp[i][j],dp[i-1][j]+1);
                }
            }
        }
        return dp[len1][len2];
    }
};
```
