# 3.代码实现
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param s string字符串 
     * @return int整型
     */
    int longestValidParentheses(string s) {
        // write code here
        int len=s.size();
        // dp[i]：以字符串 s 的第 i 个字符（0-based 索引）结尾的最长有效括号子串的长度。
        vector<int> dp(len+1,0);
        int res=0;
        for(int i=1;i<=len;i++){
            // s[i-1]为'('，无有效匹配
            if(s[i-1]=='(') dp[i]=0;
            else{
                // s[i-1]为')' 且 s[i-2]为'('，则能相邻匹配
                if(i>=2 && s[(i-1)-1]=='('){
                    dp[i]=max(dp[i-2]+2,2);
                    res=max(res,dp[i]);
                }else{
                    /* s[i-1]为')'，且s[i-2]也为')',则不能相邻匹配
                        - 因为dp[i-1]为以第i-1个字符结尾的最长有效括号子串的长度
                        - 可通过 i-dp[i-1]-1 计算最后一个不能匹配的字符属于第几个字符
                        - 若 s[(i-dp[i-1]-1)-1] 为'('，是的话则能"远距离匹配"，否则无法匹配*/
                    int last_idx=i-dp[i-1]-1;
                    if(last_idx>=1 && s[last_idx-1]=='('){
                        // 注意！！：如果能远距离匹配，则还需要加上"前面相邻匹配数dp[last_idx-1]"
                        dp[i]=(dp[i-1]+2) + dp[last_idx-1];
                        res=max(res,dp[i]);
                    }
                }
            }
        }
        return res;
    }
};
```

# 拓：使用“栈”解决该题
![](https://cdn.jsdelivr.net/gh/xianyubuxian-TXY/hexo-images/images/20260305233549697.png)
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param s string字符串 
     * @return int整型
     */
    int longestValidParentheses(string s) {
        // write code here
        int maxLen=0;
        stack<int> Stack;
        int len=s.size();
        for(int i=1;i<=len;i++){
            if(s[i-1]=='('){
                // 为'('时，“下标”入栈
                Stack.push(i);
            }else{
                // 为')'时，看栈顶下标对应字符是否为'('，是的话出栈与')'匹配，否则'(' 入栈
                if(Stack.empty() || s[Stack.top()-1]==')'){
                    // 为空，或 栈顶下标对应字符为')',则')'直接入栈
                    Stack.push(i);
                }
                else{
                    // 不为空，且为'('，则出栈进行匹配
                    Stack.pop();

                    // 栈为空：则前i个都是“格式正确的括号子串”
                    if(Stack.empty()){
                        maxLen=i;
                        continue;
                    }

                    // 栈为空：则栈顶“下标”~i 之间都是“格式正确的括号子串”，计算匹配长度
                    int last_idx=Stack.top();
                    maxLen=max(maxLen,i-last_idx);
                }
            }
        }

        return maxLen;
    }
};
```
