# 一、BM80 买卖股票的最好时机(一)
## 1.题目描述
![](https://cdn.jsdelivr.net/gh/xianyubuxian-TXY/hexo-images/images/20260307220508836.png)

## 2.代码实现
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param prices int整型vector 
     * @return int整型
     */
    int maxProfit(vector<int>& prices) {
        // write code here
        int len=prices.size();
        if(len<=1) return 0;
        int res=0;
        /*
            1.记录最小值
            2.依次向后遍历：
                - 遇到更大的，则计算“利润”，看是否能获取最大
                - 遇到更小的，则更新“最小值” -> 之后遇到更大的，一定与当前“最小值”组成获利最大
        */
        int min_val=prices[0];  // 记录最小值        
        for(int i=1;i<len;i++){
            if(prices[i]>min_val){
                res=max(res,prices[i]-min_val);
            }else{
                min_val=prices[i];
            }
        }
        return res;
    }
};
```

# 二、BM81 买卖股票的最好时机(二)
## 1.题目描述
![](https://cdn.jsdelivr.net/gh/xianyubuxian-TXY/hexo-images/images/20260307222600266.png)

## 2.代码实现
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 计算最大收益
     * @param prices int整型vector 股票每一天的价格
     * @return int整型
     */
    int maxProfit(vector<int>& prices) {
        // write code here
        int len=prices.size();
        // dp[i][0]: 前i天最大收益，且已经卖掉所有股票
        // dp[i][1]: 前i天最大收益，且还有股票未卖
        vector<vector<int>> dp(len+1,vector<int>(2,0));
        dp[1][0]=0;
        dp[1][1]-=prices[0];
        for(int i=2;i<=len;i++){
            // 1.第i天不持有股票：本就不持有股票且第i天不买股票 或 在持有股票的情况第i天下卖掉股票
            dp[i][0]=max(dp[i-1][0],dp[i-1][1]+prices[i-1]);

            // 2.第i天持有股票：本就持有股票且第i天不卖掉股票 或 在不持有股票的情况下第i天买入股票
            dp[i][1]=max(dp[i-1][1],dp[i-1][0]-prices[i-1]);
        }
        return max(dp[len][0],dp[len][1]);
    }
};
```

## 3.空间优化
- 有"2.代码实现"可知，dp[i]至于dp[i-1]有关，记录dp[i-1]即可
```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 计算最大收益
     * @param prices int整型vector 股票每一天的价格
     * @return int整型
     */
    int maxProfit(vector<int>& prices) {
        // write code here
        int len=prices.size();
        // dp[i][0]: 前i天最大收益，且已经卖掉所有股票
        // dp[i][1]: 前i天最大收益，且还有股票未卖
        int pre1=0;             // dp[i-1][0]
        int pre2=-prices[0];    // dp[i-1][1]
        for(int i=2;i<=len;i++){
            int cur1=max(pre1,pre2+prices[i-1]);
            int cur2=max(pre2,pre1-prices[i-1]);

            pre1=cur1;
            pre2=cur2;
        }
        return max(pre1,pre2);
    }
};
```

# 三、BM82 买卖股票的最好时机(三)
## 1.题目描述
![](https://cdn.jsdelivr.net/gh/xianyubuxian-TXY/hexo-images/images/20260307233616096.png)

## 2.代码实现
- **空间优化**：用变量记录用到的dp[i-1]即可
```cpp
#include <climits>
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 两次交易所能获得的最大收益
     * @param prices int整型vector 股票每一天的价格
     * @return int整型
     */
    int maxProfit(vector<int>& prices) {
        // write code here
        // dp[i][hold][k]: 前 i 天(1..n)，hold(持有/不持有)=0/1，完成 k 笔交易(0..2) 的最大收益  
        int len=prices.size();
        // 注意：不能用INT_MIN初始化，会导致溢出
        vector<vector<vector<int>>> dp(len+1,vector<vector<int>>(2,vector<int>(3,-200000)));
        dp[0][0][0] = 0; // 第0天：不持股，0笔交易，收益0
    
        for(int i=1;i<=len;i++){
            int price=prices[i-1];
            // 进行的交易次数
            for(int k=0;k<=2;k++){
                // 1.第i天不操作
                dp[i][0][k]=dp[i-1][0][k];
                dp[i][1][k]=dp[i-1][1][k];

                // 2.第i天买入股票
                dp[i][1][k]=max(dp[i][1][k],dp[i-1][0][k]-price);   // k=2买入没关系，最后不会取到

                // 3.第i天卖出股票
                if(k>=1){
                    dp[i][0][k]=max(dp[i][0][k],dp[i-1][1][k-1]+price);
                }
            }
        }

        // 最终必须不持股才是已落袋收益
        return max({dp[len][0][0], dp[len][0][1], dp[len][0][2]});
    }
};
```
