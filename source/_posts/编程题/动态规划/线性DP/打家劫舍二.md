class Solution {
public:
    // 线性挑选
    int func(vector<int>& nums,int left,int right){
        // dp[i] = 在只考虑房子 l..i 的情况下，且不能偷相邻房子的约束下，最多能偷到的钱。
        // 不偷i，则dp[i]=dp[i-1]; 偷i，则不能偷i-1,dp[i]=dp[i-2]+nums[i-1]
        int pre1=0; // dp[i-1]
        int pre2=0; // dp[i-2]
        for(int i=left; i<=right;i++){
            int cur=max(pre1,pre2+nums[i-1]);
            pre2=pre1;
            pre1=cur;
        }
        return pre1;
    }

    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param nums int整型vector 
     * @return int整型
     */
    int rob(vector<int>& nums) {
        // write code here
        int len=nums.size();
        int res=0;
        /*
        第1间、最后一间不可以同时偷，故分为2种情况
            1.如果不偷第1间：
                - 需要考虑：会不会出现最大值出现在“不偷最后一间”，需要加上“第1间”？
                    - 不需要，因为正好对应“不偷最后一间的情况”
            2.如果不偷最后一间：
            同理
        */
        
        // 不偷第1间
        res=max(res,func(nums,2,len));
        // 不偷最后一间
        res=max(res,func(nums,1,len-1));

        return res;
    }
};
