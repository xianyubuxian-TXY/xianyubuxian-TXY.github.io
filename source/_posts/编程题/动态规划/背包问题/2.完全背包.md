# 3.代码实现
```cpp
#include <iostream>
using namespace std;
#include<vector>

int main() {
    int T;
    cin>>T;
    for(int i=0;i<T;i++){
        int n,m;
        cin>>n>>m;
        vector<pair<int,int>> arr(n);  //{w,v}
        for(int j=0;j<n;j++){
            int w,v;
            cin>>w>>v;
            arr[j]={w,v};
        }

        // dp[i][j]：i种物品装容量为j的背包能带走的最大物品价值
        vector<vector<int>> dp(n+1,vector<int>(m+1,0)); 
        for(int k=1;k<=m;k++){  // 遍历“背包容量”
            for(int t=1;t<=n;t++){  // 遍历“物品数量”：表示背包容量为k时，最后一个被装入的物品
                // 不装第t个物品
                dp[t][k]=dp[t-1][k];
                if(k-arr[t-1].first>=0){
                    // 装入第t个物品
                    // ⚠️：这里必须是dp[i][j-w[i]]而不是dp[i-1][j-w[i]]
                    dp[t][k]=max(dp[t][k],dp[t][k-arr[t-1].first]+arr[t-1].second);
                }
            }
        }
        cout<<dp[n][m]<<endl;
    }
    return 0;
}
// 64 位输出请用 printf("%lld")
```
