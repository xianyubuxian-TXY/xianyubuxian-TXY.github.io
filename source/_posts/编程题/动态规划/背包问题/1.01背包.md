# 3.代码实现
```cpp
#include <iostream>
#include <utility>
using namespace std;
#include<vector>

int main() {
    int n,V;
    cin>>n>>V;

    vector<int> arrV(n+1);  // 体积
    vector<int> arrW(n+1);  // 价值

    for(int i=1;i<=n;i++){
        cin>>arrV[i]>>arrW[i];
    }

    // dp1[i][j]：表示将前i个物品装入容量为j的背包中，能够获取的最大价值（不要求装满）
    // dp1[i][j]=max(dp1[i-1][j],dp1[i-1][j-arrV[i]]+arrW[i]) ——> 对于第i个物品，不装或装
    vector<vector<int>> dp1(n+1,vector<int>(V+1,0));
    for(int i=1;i<=n;i++){
        for(int j=1;j<=V;j++){
            dp1[i][j]=dp1[i-1][j];  // 不装第i个物品

            if(j-arrV[i]>=0){
                // 装第i个物品
                dp1[i][j]=max(dp1[i][j],dp1[i-1][j-arrV[i]]+arrW[i]);
            }
        }
    }

    // dp2[i][j]：表示将前i个物品恰好装满容量为j的背包时，能够获取的最大价值（要求恰好装满）
    // dp2[i][j]=max(dp2[i-1][j],dp2[i-1][j-arrV[i]]+arrW[i]), 且 dp2[i-1][j-arrV[i]]！=-1
    vector<vector<int>> dp2(n+1,vector<int>(V+1,-1));   // -1表示无效（即无法装满）
    
    for(int i=0;i<=n;i++) dp2[i][0]=0;  // 初始化（与dp1不同点）

    for(int i=1;i<=n;i++){
        for(int j=1;j<=V;j++){
            dp2[i][j]=dp2[i-1][j];  // 不装第i个物品

            if(j-arrV[i]>=0 && dp2[i-1][j-arrV[i]]!=-1){
                // 装第i个物品
                dp2[i][j]=max(dp2[i][j],dp2[i-1][j-arrV[i]]+arrW[i]);
            }
        }
    }

    cout<<dp1[n][V]<<endl;
    if(dp2[n][V]!=-1) cout<<dp2[n][V];
    else cout<<0;

    return 0;
}
```
