# 3.代码实现
```cpp
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        deque<int> q;          // 存下标
        vector<int> ans;

        for (int i = 0; i < nums.size(); i++) {
            // 1. 维护单调递减：把队尾比当前元素小的都弹掉
            while (!q.empty() && nums[q.back()] < nums[i]) {
                q.pop_back();
            }

            // 2. 当前元素入队
            q.push_back(i);

            // 3. 删除已经不在窗口内的队头元素
            if (!q.empty() && q.front() <= i - k) {
                q.pop_front();
            }

            // 4. 窗口形成后，记录最大值
            if (i >= k - 1) {
                ans.push_back(nums[q.front()]);
            }
        }

        return ans;
    }
};
```
