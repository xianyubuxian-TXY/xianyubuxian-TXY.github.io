## （1）滑动窗口 + 动态规划

```cpp
class Solution {
public:
    int minSumOfLengths(vector<int>& arr, int target) {
        int len=arr.size();
        if(len==0) return -1;
        const int INF=1e9;

        // 滑动窗口
        list<int> Ls;
        int sum=0;  // 滑动窗口内数字之和
        int left=1,right=1; // 双指针，滑动窗口边缘（下标从1开始）
        
        // dp[i]：arr[0,i]之间和为target的最短连续子数组长度
        vector<int> dp(len+1,INF);

        // 结果
        int res=INF;

        // 对right用for循环，方便计算每个dp[i]
        for(;right<=len;right++){
            // 至少能等于dp[right-1]
            dp[right]=dp[right-1];

            Ls.push_back(arr[right-1]);   // 放入窗口
            sum+=arr[right-1];
            if(sum==target){    // 正好等于target
                int tmp_len=Ls.size();
                dp[right]=min(dp[right],tmp_len);
                // 更新res
                res=min(res,dp[left-1]+tmp_len);
            }else if(sum>target){   // 超过target，left向右收缩
                while(sum>target){
                    left++;
                    sum-=Ls.front();
                    Ls.pop_front();
                    if(sum==target){
                        int tmp_len=Ls.size();
                        dp[right]=min(dp[right],tmp_len);
                        res=min(res,dp[left-1]+tmp_len);
                    }
                }
            }
        }

        return res==INF?-1:res;
    }
};
``` 

## （2）前缀和 + 动态规划
- **使用前提**：arr[i]>=1,即是单增的，则sum也是单增的，不会出现sum相同的情况，即不会出现同一个sum对应多个index
```cpp
class Solution {
public:
    int minSumOfLengths(vector<int>& arr, int target) {
        int len=arr.size();
        const int INF=1e9;
        vector<int> dp(len+1,INF);    // dp[i]：表示[0,i] 区间何为target的长度最小数组
        int res=INF;
        int sum=0;
        unordered_map<int,int> uMap;    // <sum,index>
        for(int i=1;i<=len;i++){
            // 计算当前前缀和
            sum+=arr[i-1];
            
            // 前缀和正好为target
            if(sum==target){
                dp[i]=i;
                uMap[sum]=i;
                continue;
            }
            
            dp[i]=dp[i-1];  //先继承前面的

            // 通过前面“前缀和”为sum-target的位置
            if(uMap.count(sum-target)){
                int preIndex=uMap[sum-target];  // [preIndex,i] 为何为target的连续子数组
                int len=i-preIndex; // 计算该和为target的连续子数组长度
                res=min(res,len+dp[preIndex]);    // dp[preIndex]+len([preIndex,i]) 看是否更新res
                dp[i]=min(dp[i],len);
            }
            uMap[sum]=i;
        }
        return res==INF?-1:res;
    }
};
```
