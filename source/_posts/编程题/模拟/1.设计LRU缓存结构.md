# 2.题目分析
- **双向链表**：O(1)插入、删除
- **unordered_map<\key,Node>**：O(1)获取Node*

# 3.代码实现
## （1）使用STL库（推荐）
- 使用STL的list，避免手动实现双向链表，更可靠
- **list<pair<int,int>> + unordered_map<int,list<pair<int,int>>::iterator>**
	- 存储 **key——>迭代器** 映射

```cpp
#include <unordered_map>
class Solution {
public:
    list<pair<int,int>> ls_;    // 双向链表，存<key,val>
    unordered_map<int,list<pair<int,int>>::iterator> umap_;
    int capacity_;

 Solution(int capacity){
 // write code here
    capacity_=capacity;
 }
 
 int get(int key) {
 // write code here
    auto it=umap_.find(key);
    // 不存在，返回-1
    if(it==umap_.end()) return -1;

    // 已存在，更新位置，并返回值
    int val=umap_[key]->second;
    ls_.erase(umap_[key]);  // 从链表中删除该节点（通过迭代器删除）
    ls_.push_front({key,val});  // 再次创建并插入
    
    return val;
 }
 
 void set(int key, int value){
 // write code here
    auto it=umap_.find(key);
    if(it!=umap_.end()){
        // 已存在
        ls_.erase(umap_[key]);          // 从链表中删除（通过迭代器删除）
        ls_.push_front({key,value});    // 插入头部
        umap_[key]=ls_.begin();         // 更新umap_ 
    }else{
        // 不存在
        if(umap_.size()<capacity_){
            // 容量足够，直接插入
            ls_.push_front({key,value});
            umap_.insert({key,ls_.begin()});
        }else{
            // 容量不足,先删除最久未使用节点，再插入
            auto last=ls_.back();   
            umap_.erase(last.first);  // 通过 key 删除
            ls_.pop_back();

            // 插入头部
            ls_.push_front({key,value});
            umap_.insert({key,ls_.begin()});
        }
    }
 }
};

/**
 * Your Solution object will be instantiated and called as such:
 * Solution* solution = new Solution(capacity);
 * int output = solution->get(key);
 * solution->set(key,value);
 */
```

## （2）自己实现双向链表
- **注意**：插入新节点 与 更新已有节点位置 方法不同
	- **插入新节点**：直接插入头部即可
	- **更新已有节点位置**：需要先取出节点，再插入头部
```cpp
class Solution {
public:
    struct Node{
        int key;
        int val;
        Node* pre=nullptr;
        Node* next=nullptr;
        Node(int k,int v): key(k),val(v){}
    };

    int capacity_; // 容量
    // 手动维护双向链表（含哨兵）——> 头部：最新  尾部：最旧
    Node* head=new Node(-1,-1);  // 头节点：哨兵
    Node* tail=new Node(-1,-1);  // 尾节点：哨兵

    // 为了能O（1）内获取指定“节点指针”，配合unordered_map<key,Node*> 
    unordered_map<int,Node*> umap;
        
    // 删除指定节点
    void remove(Node* a){
        a->pre->next=a->next;
        a->next->pre=a->pre;
    }

    // 在头部插入节点
    void insertHead(Node *a){
        head->next->pre=a;
        a->next=head->next;
        head->next=a;
        a->pre=head;
    }

    // 更新节点位置
    void updateNodePos(Node* a){
        // 1.从链表中取出a
        a->next->pre=a->pre;
        a->pre->next=a->next;

        // 2.插入头部
        insertHead(a);
    }

    // 删除尾部节点
    void deleteTail(){
        Node* node=tail->pre;
        if(node==head) return;
        // 从双向链表中删除节点
        tail->pre=node->pre;
        node->pre->next=node->next;
        // 从umap中删除节点
        umap.erase(node->key);
        delete node;
    }

    Solution(int capacity){
        // write code here
        capacity_=capacity;
        head->next=tail;
        tail->pre=head;
    }
    
    int get(int key) {
    // write code here
    auto it=umap.find(key);
    if(it==umap.end()) return -1;
    else{
        // 更新节点位置
        updateNodePos(it->second);
        // 返回节点值
        return it->second->val;
    }
    }
    
    void set(int key, int value){
    // write code here
    auto it=umap.find(key);
    if(it!=umap.end()){
        // 已存在，修改值
        it->second->val=value;
        // 更新位置
        updateNodePos(it->second);
    }else{
        Node* node=new Node(key,value); // 新节点
        // 不存在
        // 1.未超过容量，直接插入
        if(umap.size()<capacity_){
            insertHead(node);
            umap.insert({key,node});
        }
        // 2.超过容量，需要弹出最久未用，再插入
        else{
            // 删除最久未用节点
            deleteTail();
            // 插入新节点
            insertHead(node);
            umap.insert({key,node});
        }
    }
    }
};

/**
 * Your Solution object will be instantiated and called as such:
 * Solution* solution = new Solution(capacity);
 * int output = solution->get(key);
 * solution->set(key,value);
 */
```
