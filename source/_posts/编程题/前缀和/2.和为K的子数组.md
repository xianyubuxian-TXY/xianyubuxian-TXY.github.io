# 3.代码实现
```cpp
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int len=nums.size();
        // <sum,times>
        unordered_map<int,int> uMap;
        
        // uMap[0]=1：专门处理 sum==k 的情况
        uMap[0]=1;

        int res=0;
        int sum=0;
        for(int i=0;i<len;i++){
            sum+=nums[i];
            if(uMap.count(sum-k)){
                res+=uMap[sum-k];
            }
            if(uMap.count(sum)) uMap[sum]++;
            else uMap[sum]=1;
        }


        return res;
    }
};
```
