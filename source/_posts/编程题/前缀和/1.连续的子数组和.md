# 2.题目分析
- 很容易想到**“前缀和”+双循环**，但数组长度为10^5,肯定会超时

# 3.代码实现
```cpp
class Solution {
public:
    bool checkSubarraySum(vector<int>& nums, int k) {
        int len=nums.size();
        // 存储：<余数，下标>
        unordered_map<int,int> Map; 
        // 将Map[0]设置为0 ——> 处理前缀和正好为k整数倍的情况
        Map[0]=0;

        // 前缀和：只需要记录当前前缀和即可
        int prefixSum=0;
        for(int i=1;i<=len;i++){
            prefixSum+=nums[i-1];

            int mod=prefixSum%k;

            if(Map.count(mod)){
                // 不是第一次出现，且距离大于2
                if(i-Map[mod] >=2) return true;
            }else{
                Map[mod]=i;
            }

        }
        return false;
    }
};
```
