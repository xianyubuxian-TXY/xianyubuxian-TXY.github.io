## （1）多源DFS代码实现
```cpp
class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {

        queue<pair<int,int>> Que;   // 队列，用于多源BFS
        int n=mat.size();
        int m=mat[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                // 先将所有为0的位置加入到“队列中”
                if(mat[i][j]==0) Que.push({i,j});
                else{
                    // 对于元素为'1'的位置，置为-1：表示还未被访问过
                    mat[i][j]=-1;
                }
            }
        }

        int dirs[4][2]={{-1,0},{1,0},{0,-1},{0,1}}; // 上下左右

        while(!Que.empty()){
            // 获取队列大小，一次性遍历（多源），每次遍历就是一层
            // 不能用while(!Que.empty()),因为中间会增加元素到Que中
            int size=Que.size(); 
            for(int i=0;i<size;i++){
                auto it=Que.front();  // 取出元素
                Que.pop();

                // 获取当前元素坐标，方便之后遍历四个方向
                int row=it.first;
                int col=it.second;

                // 遍历四个方向
                for(auto d:dirs){
                    int nrow=row+d[0];
                    int ncol=col+d[1];

                    // 位置有效，且未被访问（即mat[nx][ny]==-1）
                    // 因为是多源dfs，同一层同时向四周扩展，所以如果这个过程扩展到同一个位置，距离是一样的，同时也保证了不会重复将一个位置加入到Que中
                    if(nrow>=0 && nrow<n && ncol>=0 && ncol<m && mat[nrow][ncol]==-1){
                        mat[nrow][ncol]=mat[row][col]+1;    // 更新距离，同时也可以避免被重复加入Que
                        Que.push({nrow,ncol});  // 加入Que 
                    }
                }
            }
        }
        return mat;

    }
};
```

## （2）动态规划代码实现
```cpp
class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        const int INF=1000000;

        // 直接用mat当dp即可，正好为0的位置，对应的dp值也为0

        // 第一次遍历：从左上角开始 ——> 遍历过程类似向右、下移动
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(mat[i][j]!=0){   // 建议用 "!=0",而不是"==1",虽然第一次遍历可以，但为了和第二次遍历统一
                    // 如果左、上方相邻不为0，则设置为INF，先不可达
                    int up=(i-1>=0)?mat[i-1][j]+1:INF;   // 看能否从上方来
                    int left=(j-1>=0)?mat[i][j-1]+1:INF;  // 看能否从左方来
                    mat[i][j]=min(up,left);
                }
            }
        }

        // 第二次遍历：从右下角开始
        for(int i=n-1;i>=0;i--){
            for(int j=m-1;j>=0;j--){
                // if(mat[i][j]==1)  ❌ 错！
                if(mat[i][j]!=0){   // ⚠️ 一定要用"!=0",不能用"==1",因为第一次遍历可能修改了原本为"1"的值
                    int down=(i+1<n)?mat[i+1][j]+1:INF;
                    int right=(j+1<m)?mat[i][j+1]+1:INF;
                    // ⚠️ 注意！！：这里一定要与mat[i][j]自己比，避免覆盖第一次遍历的结果
                    mat[i][j]=min(mat[i][j],min(down,right));
                }
            }
        }

        return mat;
    }
};
```
