# 3.代码实现
```cpp
class Solution {
public:
    int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
        // 方便后面判断一个单词是否存在与wordList
        unordered_set<string> dict(wordList.begin(),wordList.end());
    
        // endWord 不在字典中
        if(!dict.count(endWord)) return 0;

        // 如果beginWord在字典中存在，则去掉，避免重复加入队列
        if(dict.count(beginWord)) dict.erase(beginWord);

        queue<string> Que;  // 队列，用于BFS
        Que.push(beginWord);

        int len=1;  //序列长度
        while(!Que.empty()){
            int size=Que.size();    // 获取当前层的元素个数
            for(int i=0;i<size;i++){
                // 获取元素
                string s=Que.front();
                Que.pop();

                // 如果当前元素已经是endWord，直接放回
                if(s==endWord) return len;

                // 尝试将该字符串的每个位置都依次替换为26个字母，看是否在wordList中
                // 注意：只能替换一个，所以替换后需要“回溯”
                for(int j=0;j<s.size();j++){
                    int oldchar=s[j];   // 保存原来位置的字符
                    for(int k=0;k<26;k++){
                        s[j]='a'+k;

                        // 看是否在wordList中
                        if(dict.count(s)){
                            Que.push(s);
                            // ⚠️注意：加入队列后，需要将s从dict中去掉，避免之后反复加入
                            dict.erase(s);
                        }
                    }
                    
                    s[j]=oldchar;   // 回溯
                }
            }
            len++;  // 序列长度+1
        }
        return 0;
    }
};
```
